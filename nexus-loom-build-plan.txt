STEP 02	Supabase Init & Auth Layer
📅  Day 1 — Afternoon	BACKEND

Sets up Supabase client, creates the auth repository, and wires secure token storage. No Firebase anywhere.

⚡  PROMPT — Supabase Setup
In main.dart, initialize Supabase before runApp() — NOT Firebase.

  await Supabase.initialize(
    url: const String.fromEnvironment('SUPABASE_URL'),
    anonKey: const String.fromEnvironment('SUPABASE_ANON_KEY'),
  );

Wrap MaterialApp with ProviderScope (Riverpod).
Set the initial route using go_router based on auth state:
  - If Supabase.instance.client.auth.currentSession != null → '/dashboard'
  - Otherwise → '/login'

Create lib/data/repositories/auth_repository.dart with these methods:
  Future<AuthResponse> signIn(String email, String password)
  Future<AuthResponse> signUp(String email, String password, String displayName)
  Future<void> signOut()
  Stream<AuthState> get authStateChanges
  User? get currentUser

Use flutter_secure_storage to persist and retrieve the refresh token.
The Supabase client handles JWT refresh automatically — do not add any
Firebase Auth code. There is no Firebase in this project.

⚡  PROMPT — App Router
Create lib/core/router/app_router.dart using go_router ^13.

Define these routes:
  /login         → LoginScreen
  /dashboard     → DashboardScreen (with ShellRoute for bottom nav)
  /dashboard/tasks → TasksScreen
  /dashboard/notes → NotesScreen
  /dashboard/alerts → AlertsScreen
  /task/:id      → TaskDetailScreen
  /task/:id/section/:sectionId → SectionDetailScreen
  /note/:id      → NoteDetailScreen

ShellRoute for /dashboard wraps screens with a shared bottom nav bar
(Home, Tasks, Notes, Alerts) using AppColors.primary for selected icon.

Add redirect guard: if no Supabase session → redirect to /login.
Export router as final appRouter = GoRouter(...) provider.

  Stream<List<ItemModel>> watchBySpace(String spaceId) =>
    _isar.db.itemModels.filter().spaceIdEqualTo(spaceId).watch();

  Stream<List<BlockModel>> watchBlocks(String itemId) =>
    _isar.db.blockModels.filter().itemIdEqualTo(itemId)
      .sortByOrderIndex().watch();

  Future<void> saveItem(ItemModel item) async {

    item.updatedAt = DateTime.now();
    await _isar.db.writeTxn(() => _isar.db.itemModels.put(item));
  }

  Future<void> saveBlock(BlockModel block) async {
    block.syncStatus = 'pending';
    block.updatedAt = DateTime.now();
    await _isar.db.writeTxn(() => _isar.db.blockModels.put(block));
  }

      .filter().idEqualTo(itemId).findFirst();

    item.syncStatus = 'pending';
    item.updatedAt = DateTime.now();

  }

class SyncManager {
  static final SyncManager instance = SyncManager._();
  SyncManager._();

  StreamSubscription? _connectivitySub;
  bool _isSyncing = false;

  // Call once after Isar and Supabase are initialized
  void start() {
    // Listen to connectivity changes via connectivity_plus
    _connectivitySub = Connectivity().onConnectivityChanged.listen((result) {
      final isOnline = result != ConnectivityResult.none;
      if (isOnline) _runSyncCycle();
    });
    // Also try sync on startup
    _runSyncCycle();
  }

  Future<void> triggerSync() async => _runSyncCycle();

  Future<void> _runSyncCycle() async {
    if (_isSyncing) return;
    _isSyncing = true;
    try {
      await _pullRemoteChanges();  // Pull first
      await _pushPendingItems();   // Then push
      await _pushPendingBlocks();
    } finally {
      _isSyncing = false;
    }
  }

  // PULL: fetch all items updated after lastSyncedAt
  Future<void> _pullRemoteChanges() async {
    final lastSync = await SecureStorageService.getLastSyncedAt();
    final client = Supabase.instance.client;

    final items = await client.from('items')
      .select().gt('updated_at', lastSync.toIso8601String());
    for (final row in items) {
      final existing = await IsarService.instance.db.itemModels
        .filter().idEqualTo(row['id']).findFirst();
      final remote = ItemModel.fromJson(row);
      // Only overwrite local if remote is newer
      if (existing == null || remote.updatedAt.isAfter(existing.updatedAt)) {
        remote.syncStatus = 'synced';
        await IsarService.instance.db.writeTxn(
          () => IsarService.instance.db.itemModels.put(remote));
      }
    }
    // Same pattern for item_blocks

    await SecureStorageService.setLastSyncedAt(DateTime.now());
  }

  // PUSH: upsert all pending items to Supabase
  Future<void> _pushPendingItems() async {
    final pending = await IsarService.instance.db.itemModels
      .filter().syncStatusEqualTo('pending').findAll();
    for (final item in pending) {
      try {
        await Supabase.instance.client.from('items')
          .upsert(item.toJson(), onConflict: 'id');
        item.syncStatus = 'synced';
        await IsarService.instance.db.writeTxn(
          () => IsarService.instance.db.itemModels.put(item));
      } catch (e) {
        // Keep as pending, retry next cycle
        debugPrint('Sync error: $e');
      }
    }
  }

  Future<void> _pushPendingBlocks() async {
    // Same pattern as _pushPendingItems but for block_models table
  }

  void dispose() => _connectivitySub?.cancel();
}

class RealtimeService {
  final _subscriptions = <RealtimeChannel>[];

  // Call after user joins a space
  void subscribeToSpace(String spaceId) {
    // Subscribe to item changes in this space
    final itemChannel = Supabase.instance.client
      .channel('space_items_$spaceId')
      .onPostgresChanges(
        event: PostgresChangeEvent.all,
        schema: 'public',
        table: 'items',
        filter: PostgresChangeFilter(
          type: PostgresChangeFilterType.eq,
          column: 'space_id',
          value: spaceId,
        ),
        callback: (payload) => _handleItemChange(payload),
      )
      .subscribe();

    // Subscribe to block changes via item_blocks
    final blockChannel = Supabase.instance.client
      .channel('space_blocks_$spaceId')
      .onPostgresChanges(
        event: PostgresChangeEvent.all,
        schema: 'public',
        table: 'item_blocks',
        callback: (payload) => _handleBlockChange(payload),
      )
      .subscribe();

    _subscriptions.addAll([itemChannel, blockChannel]);
  }

  // Subscribe to this user's notification stream
  void subscribeToNotifications(String userId) {
    Supabase.instance.client
      .channel('user_notifications_$userId')
      .onPostgresChanges(
        event: PostgresChangeEvent.insert,
        schema: 'public',
        table: 'notifications',
        filter: PostgresChangeFilter(
          type: PostgresChangeFilterType.eq,
          column: 'user_id',
          value: userId,
        ),
        callback: (payload) => _handleNotification(payload),
      )
      .subscribe();
  }

  Future<void> _handleItemChange(PostgresChangePayload payload) async {
    final data = payload.newRecord;
    if (data.isEmpty) return;
    final item = ItemModel.fromJson(data);
    item.syncStatus = 'synced';
    // Write to Isar — Riverpod providers watching Isar will rebuild automatically
    await IsarService.instance.db.writeTxn(
      () => IsarService.instance.db.itemModels.put(item));
  }

  Future<void> _handleBlockChange(PostgresChangePayload payload) async {
    final block = BlockModel.fromJson(payload.newRecord);
    block.syncStatus = 'synced';
    await IsarService.instance.db.writeTxn(
      () => IsarService.instance.db.blockModels.put(block));
  }

  Future<void> _handleNotification(PostgresChangePayload payload) async {
    final notif = NotificationModel.fromJson(payload.newRecord);
    await IsarService.instance.db.writeTxn(
      () => IsarService.instance.db.notificationModels.put(notif));
    // Riverpod notificationCountProvider rebuilds automatically
  }

  void unsubscribeAll() {
    for (final sub in _subscriptions) {
      Supabase.instance.client.removeChannel(sub);
    }
    _subscriptions.clear();
  }
}

class LocalNotificationService {
  static final _plugin = FlutterLocalNotificationsPlugin();

  static Future<void> init() async {
    const androidInit = AndroidInitializationSettings('@mipmap/ic_launcher');
    const iosInit = DarwinInitializationSettings(
      requestAlertPermission: true,
      requestBadgePermission: true,
      requestSoundPermission: true,
    );
    await _plugin.initialize(
      const InitializationSettings(android: androidInit, iOS: iosInit),
      onDidReceiveNotificationResponse: _onTap,
    );
  }

  // Schedule a local reminder for a task — NO network call needed
  static Future<void> scheduleReminder({
    required int id,
    required String title,
    required String body,
    required DateTime scheduledAt,
  }) async {
    await _plugin.zonedSchedule(
      id,
      title,
      body,
      tz.TZDateTime.from(scheduledAt, tz.local),
      const NotificationDetails(
        android: AndroidNotificationDetails('reminders', 'Task Reminders'),
        iOS: DarwinNotificationDetails(),
      ),
      androidScheduleMode: AndroidScheduleMode.exactAllowWhileIdle,
      uiLocalNotificationDateInterpretation:
        UILocalNotificationDateInterpretation.absoluteTime,
    );
  }

  static Future<void> cancelReminder(int id) =>
    _plugin.cancel(id);

  static void _onTap(NotificationResponse response) {
    // Navigate to item using go_router navigatorKey
    final itemId = response.payload;
    if (itemId != null) {
      appRouter.push('/task/$itemId');
    }
  }
}

class StorageRepository {
  final _client = Supabase.instance.client;

  // Upload image to Supabase Storage — NOT Firebase Storage
  Future<String?> uploadBlockImage({
    required String spaceId,
    required File imageFile,
  }) async {
    // Compress locally first
    final compressed = await FlutterImageCompress.compressWithFile(
      imageFile.absolute.path,
      quality: 75,
      minWidth: 1200,
      minHeight: 1200,
    );
    if (compressed == null) return null;

    final fileName = '${const Uuid().v4()}.jpg';
    final path = 'spaces/$spaceId/images/$fileName';

    await _client.storage
      .from('space-images')
      .uploadBinary(path, compressed,
        fileOptions: const FileOptions(contentType: 'image/jpeg'));

    return _client.storage.from('space-images').getPublicUrl(path);
  }
}
